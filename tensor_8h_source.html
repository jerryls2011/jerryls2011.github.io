187  template <class ContextForCopy>
188  void Extend(TIndex num, float growthPct, ContextForCopy* context) {
189  CAFFE_ENFORCE_GE_WITH_CALLER(dims_.size(), 1);
190  auto newDims = dims_;
191  newDims[0] += num;
192  if (!data_) {
193  Resize(newDims);
194  return;
195  }
196  auto newSize = std::accumulate(
197  newDims.begin(),
198  newDims.end(),
199  static_cast<TIndex>(1),
200  std::multiplies<TIndex>());
201  if (newSize * meta_.itemsize() <= capacity_) {
202  dims_ = newDims;
203  size_ = newSize;
204  return;
205  }
206  auto newCapacity = dims_;
207  newCapacity[0] = std::max<size_t>(
208  newDims[0], std::ceil(dims_[0] * (growthPct + 100) / 100));
209  Reserve(newCapacity, context);
210  dims_ = newDims;
211  size_ = newSize;
212  }

214  template <class T, class ContextForCopy>
215  void Reserve(const std::vector<T>& newCapacity, ContextForCopy* context) {
216  auto newSize = std::accumulate(
217  newCapacity.begin(),
218  newCapacity.end(),
219  static_cast<TIndex>(1),
220  std::multiplies<TIndex>());
221  if (newSize * meta_.itemsize() <= capacity_) {
222  return;
223  }
224  auto oldData = std::move(data_);
225  auto oldSize = size_;
226  auto oldDims = dims_;
227  Resize(newCapacity);
228  auto* newData = raw_mutable_data(meta_);
229  context->template CopyItems<ContextForCopy, ContextForCopy>(
230  meta_, oldSize, oldData.get(), newData);
231  dims_ = oldDims;
232  size_ = oldSize;
233  reserved_ = true;
234  }

268  template <typename... Ts>
269  void Resize(Ts... dim_source) {
270  bool size_changed = SetDims(dim_source...);
271  if (size_changed) {
272  // If needed, we will free the data. the next mutable_data() call
273  // will create the data storage.
274  int64_t new_size = size_ * meta_.itemsize();
275  bool reset_tensor = false;
276  if (reserved_) {
277  // If tensor is reserved then don't claim its memeory unless capacity_
278  // is smaller than new size
279  reset_tensor = capacity_ < new_size;
280  } else {
281  reset_tensor = capacity_ < new_size || !FLAGS_caffe2_keep_on_shrink ||
282  capacity_ - new_size > FLAGS_caffe2_max_keep_on_shrink_memory;
283  }
284 
285  if (reset_tensor) {
286  FreeMemory();
287  }
288  }
289  }

490  inline void* raw_mutable_data(const TypeMeta& meta) {
491  // For 0-size tensors it's fine to return any pointer (including nullptr)
492  if (meta_ == meta && (data_.get() || size_ == 0)) {
493  return data_.get();
494  } else {
495  meta_ = meta;
496  CAFFE_ENFORCE_WITH_CALLER(
497  size_ >= 0,
498  "Tensor is not initialized. You probably need to call Resize() "
499  "before calling mutable_data()");
500  if (size_ == 0) {
501  return data_.get();
502  }
503  if (meta.ctor()) {
504  // For types that need placement new, we will call it, as well as
505  // making sure that when the data is freed, it calls the right
506  // destruction procedure.
507  auto size = size_;
508  auto dtor = meta_.dtor();
509  auto ptr_and_deleter = Context::New(size_ * meta_.itemsize());
510  auto deleter = std::move(ptr_and_deleter.second);
511  data_.reset(
512  ptr_and_deleter.first, [size, dtor, deleter](void* ptr) -> void {
513  dtor(ptr, size);
514  deleter(ptr);
515  });
516  meta_.ctor()(data_.get(), size_);
517  } else {
518  // For fundamental type, new and delete is easier.
519  auto ptr_and_deleter = Context::New(size_ * meta_.itemsize());
520  data_.reset(ptr_and_deleter.first, std::move(ptr_and_deleter.second));
521  }
522  capacity_ = size_ * meta_.itemsize();
523  return data_.get();
524  }
525  }

790 template <class T>
791 void TensorPrinter::Print(const Tensor<CPUContext>& tensor) {
792  std::stringstream values_stream;
793  // One most likely doesn't want to print int64-number of items for visual
794  // inspection, so we cast down to int here.
795  int total_count = std::min(tensor.size(), TIndex(limit_));
796  const T* tensor_data = tensor.template data<T>();
797  for (int i = 0; i < total_count - 1; ++i) {
798  values_stream << tensor_data[i] << ",";
799  }
800  // We do not add a comma after the last item.
801  values_stream << tensor_data[total_count - 1];
802  if (to_file_) {
803  (*log_file_) << MetaStr(tensor) << values_stream.str() << std::endl;
804  } else {
805  // Log to console.
806  LOG(INFO) << MetaStr(tensor) << values_stream.str();
807  }
