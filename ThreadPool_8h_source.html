36 template <typename T>
37 struct AllocAligned {
38  // Allocate a T aligned at an `align` byte address
39  template <typename... Args>
40  static T* alloc(size_t align, Args&&... args) {
41  void* p = nullptr;
42 // FIXME: we should just be able to use std::align
43 #if !defined(__ANDROID__)
44  posix_memalign((void**)&p, align, sizeof(T));
45 #else
46  p = memalign(align, sizeof(T));
47 #endif
48 
49  if (p) {
50  return new (p) T(std::forward<Args>(args)...);
51  }
52 
53  return nullptr;
54  }
55 
56  // Free a T previously allocated via AllocAligned<T>::alloc()
57  static void release(T* p) {
58  if (p) {
59  p->~T();
60  free((void*)p);
61  }
62  }
63 };
