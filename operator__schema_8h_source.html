308  static OpSchema& NewSchema(
309  const string& key, const string& file, const int line) {
310  auto& m = map();
311  if (m.count(key)) {
312  const auto& schema = m[key];
313  std::ios_base::Init init;
314  std::cerr << "Trying to register schema with name "
315  << key << " from file " << file << " line " << line
316  << ", but it is already registered from file "
317  << schema.file() << " line " << schema.line();
318  abort();
319  }
320  m.emplace(std::make_pair(key, OpSchema(file, line)));
321  return m[key];
322  }
