13 class NetBuilder(object):
14  """
15  Scope-driven mechanism for building nets, loops and conditional blocks.
16  Example:
17  from caffe2.python.net_builder import NetBuilder, ops
18  with NetBuilder() as nb:
19  c = ops.Const(5)
20  d = ops.Const(0)
21  with ops.loop():
22  ops.stop_if(ops.LE([c, ops.Const(0)]))
23  ops.Add([c, ops.Const(-1)], [c])
24  with ops.If(ops.GE([c, ops.Const(3)])):
25  ops.Add([d, ops.Const(10)])
26  ops.Print(c, [])
27  ops.Print(d, [])
28  step = core.to_execution_step(nb)
29  """
30  def __init__(self, name=None, _stop_blob_required=False,
31  _stop_blob=None, _fullname=None):
32  nb = NetBuilder.current(required=False)
33  assert not _fullname or not name, 'Cannot set both _fullname and name'
34  self.name = _fullname or '/'.join(
35  n for n in (nb.name if nb else None, name) if n
36  )
37  self._frozen = False
38  self._current_net = None
39  self._children = []
40  self._stop_blob = _stop_blob
41  self._stop_blob_required = _stop_blob_required

43  def stop_blob(self):
44  """
45  Returns the BlobReference to the stop_blob of this NetBuilder.
46  If one is not yet available, creates one.
47  This function assumes that the stop_blob() will be used immediatelly
48  in the current net, so it doesn't initialize it if the current net is
49  the first of the builder.
50  """
51  if self._stop_blob is None:
52  net = self.current_net()
53  self._stop_blob = core.BlobReference(
54  net.NextName('stop_blob'), net=net)
55  if self._current_net != self._children[0]:
56  self._children.insert(0, core.Net('stop_blob_init'))
57  self._children[0].Const(False, blob_out=self._stop_blob)
58  return self._stop_blob
59 
60  def stop_if(self, blob):
61  ops.Copy(blob, self.stop_blob())
62  self._current_net = None

110  def net(self, net=None, name=None):
111  """
112  Retrieves the current net, or add a new net to the builder.
113  Args:
114  net: If provided, add the given net to the active builder.
115  Else, returns the current Net or creates a new one as needed.
116  name: if provided, creates a new Net with given name and makes
117  it the new current net of the active builder. Cannot
118  be provided if net is provided.
119  """
120  assert name is None or net is None, (
121  'Cannot provide both `net` and `name`.')
122  if net is not None:
123  NetBuilder.current().add(net)
124  return net
125  return NetBuilder.current().current_net(name=name)

152  """
153  Stop execution of the current execution step.
154  Example:
155  ops.Print(a, 0)
156  ops.stop()

162  def stop_if(self, blob):
163  """
164  Stop execution of the current execution step if the
165  condition `blob` is met.
166  Example:
167  ops.Print(a, 0)
168  ops.stop_if(ops.LE([x, ops.Const(0)]))
169  ops.Print(b, 0)
170  In the example, 'b' will only be printed if the value of scalar
171  tensor 'x' lower or equal to 0.
172  """
173  return NetBuilder.current().stop_if(blob)

175  def loop(self, iters=None, name=None):
176  """
177  Creates a NetBuilder that will execute in a loop as the next step of
178  the current NetBuilder. If `iters` is provided, the loop will execute
179  for `iters` iterations and then stop. `iters` can be a constant or a
180  BlobReference. If `iters` is not provided, the loop will execute
181  until `ops.stop` or `ops.stop_if` is called.
182  Examples:
183  a = ops.Const(5)
184  with ops.loop():
185  ops.stop_if(ops.LE([a, ops.Const(0)]))
186  ops.Print(a, 0)
187  ops.Add([a, ops.Const(-1)], [a])
188  Above, 'a' will be printed 5 times, with values 5 to 1.
189 
190  with ops.loop(10) as loop:
191  ops.LogInfo(loop.iter())
192  This will print the numbers from 0 to 9.
193 
194  x = ops.Add([ops.Const(10), ops.Const(10)])
195  with ops.loop(x) as loop:
196  ops.LogInfo(loop.iter())
197  This will print the numbers from 0 to 19.
198  """
199  return NetBuilder.current().add(_Loop(iters, name=name))

201  def stop_guard(self, has_stopped_blob=None, name=None):
202  """
203  Creates a NetBuilder that will execute once as the next step of the
204  current NetBuilder. After execution, a bool tensor will indicate
205  whether the inner execution was halted with `stop` or `stop_if`.
206  Example:
207  a = ops.Const(True)
208  with ops.stop_guard() as sg1:
209  ops.stop_if(a)
210  ops.Print(ops.Const('did not stop'))
211  b = ops.Const(False)
212  with ops.stop_guard() as sg2:
213  ops.stop_if(b)
214  ops.Print(ops.Const('did not stop'))
215  ops.Print(sg1.has_stopped(), [])
216  ops.Print(sg2.has_stopped(), [])
217  In the example, 'did not stop' will be printed once,
218  followed by True and False.
219  """
220  return NetBuilder.current().add(
221  _StopGuard(has_stopped_blob=has_stopped_blob, name=name))

223  def If(self, cond, name=None):
224  """
225  Creates a NetBuilder that will execute once as the next step of the
226  current NetBuilder if the blob `cond` is True.
227  Example:
228  with ops.If(ops.Const(True)):
229  ops.Print(ops.Const('Will print'))
230  with ops.If(ops.Const(False)):
231  ops.Print(ops.Const('Wont print'))
232  The example will print 'Will print' once.
233  """
234  return NetBuilder.current().add(_RunIf(cond, name=name))

236  def task_init(self):
237  """
238  Defines operations that will be executed once at task startup.
239  Useful when implementing processors, that don't have access to the Task
240  top-level structure.
241 
242  This setup will be run only once, even if multiple instances of the task
243  will run in parallel. For instance-local initialization, use
244  `task_instance_init` instead.
245 
246  Example:
247  def my_processor(rec):
248  with ops.task_init():
249  one = ops.Const(1)
250  two = ops.Const(1)
251  return Tuple(
252  ops.Add(rec[0](), zero), ops.Add(rec[1](), two))
253  """
254  setup = _SetupBuilder(_SetupBuilder.INIT)
255  self.net().add_attribute(Task.TASK_SETUP, setup)
256  return setup
