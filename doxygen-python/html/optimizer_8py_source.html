72  def build_lr(self, net, param_init_net, base_learning_rate,
73  learning_rate_blob=None, policy="fixed",
74  iter_val=0, **kwargs):
75  if learning_rate_blob is None:
76  learning_rate_blob = self.get_lr_blob_name()
77  if not param_init_net.BlobIsDefined(_OPTIMIZER_ITERATION_NAME):
78  # Add training operators.
79  with core.DeviceScope(core.DeviceOption(caffe2_pb2.CPU)):
80  iteration = param_init_net.ConstantFill(
81  [], _OPTIMIZER_ITERATION_NAME, shape=[1],
82  value=iter_val,
83  dtype=core.DataType.INT64)
84  iter_mutex = param_init_net.CreateMutex([], ["iteration_mutex"])
85  net.AtomicIter([iter_mutex, iteration], [iteration])
86  else:
87  iteration = param_init_net.GetBlobRef(_OPTIMIZER_ITERATION_NAME)
88 
89  if not net.BlobIsDefined(learning_rate_blob):
90  # There is one interesting thing here: since we are minimizing, we are
91  # doing "descent" so the learning rate is set to be negative.
92  lr = net.LearningRate(
93  [iteration],
94  learning_rate_blob,
95  base_lr=-base_learning_rate,
96  policy=policy,
97  **kwargs
98  )
99  else:
100  lr = net.GetBlobRef(learning_rate_blob)
101  return lr, iteration

112  def get_auxiliary_parameters(self):
113  """Returns a list of auxiliary parameters.
114 
115  Returns:
116  aux_params: A namedtuple, AuxParams.
117 
118  aux_params.local stores a list of blobs. Each blob is a local
119  auxiliary parameter. A local auxiliary parameter is a parameter in
120  parallel to a learning rate parameter. Take adagrad as an example,
121  the local auxiliary parameter is the squared sum parameter, because
122  every learning rate has a squared sum associated with it.
123 
124  aux_params.shared also stores a list of blobs. Each blob is a shared
125  auxiliary parameter. A shared auxiliary parameter is a parameter
126  that is shared across all the learning rate parameters. Take adam as
127  an example, the iteration parameter is a shared parameter, because
128  all the learning rates share the same iteration parameter.
129  """
130  return self._aux_params

145 class SgdOptimizer(Optimizer):
146  def __init__(self, base_learning_rate=0.01, policy='fixed',
147  momentum=0.0, nesterov=1, sparse_dedup_aggregator=None,
148  **kwargs):
149  super(SgdOptimizer, self).__init__()
150  self.base_learning_rate = base_learning_rate
151  self.policy = policy
152  self.momentum = momentum
153  self.nesterov = nesterov
154  self.sparse_dedup_aggregator = sparse_dedup_aggregator
155  self.init_kwargs = kwargs
156 
157  def _run(self, net, param_init_net, param_info):
158  param = param_info.blob
159  grad = param_info.grad
160  if self.base_learning_rate == 0:
161  return
162  assert self.base_learning_rate > 0
163 
164  # We need negative sign for LR when used directly with WeightedSum
165  # below.
166  lr_sign = -1 if self.momentum else 1
167  lr, _ = self.build_lr(
168  net, param_init_net,
169  base_learning_rate=self.base_learning_rate * lr_sign,
170  policy=self.policy,
171  **(self.init_kwargs)
172  )
173 
174  dev = scope.CurrentDeviceScope()
175  if dev is None:
176  dev = core.DeviceOption(caffe2_pb2.CPU)
177 
178  # Each GPU/CPU must have its own ONE blob, thus modify the name
179  # to include device information.
180  ONE = param_init_net.ConstantFill(
181  [],
182  "ONE_{}_{}".format(dev.device_type, dev.cuda_gpu_id),
183  shape=[1],
184  value=1.0
185  )
186 
187  self._aux_params.shared.append(ONE)
188 
189  if self.momentum > 0:
190  momentum_data = param_init_net.ConstantFill(
191  param, str(param) + "_momentum", value=0.)
192  self._aux_params.local.append(momentum_data)
193 
194  if isinstance(grad, core.GradientSlice):
195  assert self.momentum == 0., "Doesn't support momentum for sparse"
196  grad = self.dedup(net, self.sparse_dedup_aggregator, grad)
197  net.ScatterWeightedSum(
198  [param, ONE, grad.indices, grad.values, lr],
199  param
200  )
201  else:
202  if self.momentum > 0.:
203  net.MomentumSGDUpdate(
204  [grad, momentum_data, lr, param],
205  [grad, momentum_data, param],
206  momentum=self.momentum,
207  nesterov=self.nesterov)
208  else:
209  coeff = lr
210 
211  net.WeightedSum(
212  [param, ONE, grad, coeff],
213  param
214  )
215 
216  def scale_learning_rate(self, scale):
217  self.base_learning_rate *= scale
218  return

481 def get_param_device(param_name, grad, param_to_device=None, default_device=None):
482  device = default_device
483  param_to_device = param_to_device or {}
484  # We first check if parameter's device has been inferred. If not,
485  # we check the gradient. This can happen if parameter is not output
486  # by any blob but created by a FetchBlob.
487  if param_name in param_to_device:
488  device = param_to_device[param_name]
489  else:
490  if isinstance(grad, core.GradientSlice):
491  grad = grad
492  if str(grad.values) in param_to_device:
493  device = param_to_device[str(grad.values)]
494  elif str(grad.indices) in param_to_device:
495  device = param_to_device[str(grad.indices)]
496  else:
497  grad_name = str(grad)
498  if grad_name in param_to_device:
499  device = param_to_device[grad_name]
500 
501  assert device is not None,\
502  "Cannot infer device for {}: no op creates it".format(param_name)
503  return device

506 def _build(model, optimizer, weights_only=False, use_param_info_optim=True):
507  param_to_device = _get_param_to_device(model)
508 
509  # Validate there are no duplicate params
510  model.Validate()
511 
512  # Call optimizer for each param
513  for param_info in model.GetOptimizationParamInfo():
514  if weights_only:
515  if param_info.blob not in model.weights:
516  continue
517  param_name = str(param_info.blob)
518 
519  device = get_param_device(param_name, param_info.grad, param_to_device)
520 
521  with core.DeviceScope(device):
522  if param_info.optimizer and use_param_info_optim:
523  param_info.optimizer(model.net, model.param_init_net, param_info)
524  else:
525  optimizer(model.net, model.param_init_net, param_info)
526  return optimizer
