269

287  )),

445  def __add__(self, other):
446  """
447  Allows to merge fields of two schema.Struct using '+' operator.
448  If two Struct have common field names, the merge is conducted
449  recursively. Here are examples:
450 
451  Example 1
452  s1 = Struct(('a', Scalar()))
453  s2 = Struct(('b', Scalar()))
454  s1 + s2 == Struct(
455  ('a', Scalar()),
456  ('b', Scalar()),
457  )
458 
459  Example 2
460  s1 = Struct(
461  ('a', Scalar()),
462  ('b', Struct(('c', Scalar()))),
463  )
464  s2 = Struct(('b', Struct(('d', Scalar()))))
465  s1 + s2 == Struct(
466  ('a', Scalar()),
467  ('b', Struct(
468  ('c', Scalar()),
469  ('d', Scalar()),
470  )),
471  )
472  """
473  if not isinstance(other, Struct):
474  return NotImplemented
475 
476  children = OrderedDict(self.get_children())
477  for name, right_field in other.get_children():
478  if name not in children:
479  children[name] = right_field
480  continue
481  left_field = children[name]
482  children[name] = left_field + right_field
483 
484  return Struct(*(viewitems(children)))

486  def __sub__(self, other):
487  """
488  Allows to remove common fields of two schema.Struct from self by
489  using '-' operator. If two Struct have common field names, the
490  removal is conducted recursively. If a child struct has no fields
491  inside, it will be removed from its parent. Here are examples:
492 
493  Example 1
494  s1 = Struct(
495  ('a', Scalar()),
496  ('b', Scalar()),
497  )
498  s2 = Struct(('a', Scalar()))
499  s1 - s2 == Struct(('b', Scalar()))
500 
501  Example 2
502  s1 = Struct(
503  ('b', Struct(
504  ('c', Scalar()),
505  ('d', Scalar()),
506  ))
507  )
508  s2 = Struct(
509  ('b', Struct(('c', Scalar()))),
510  )
511  s1 - s2 == Struct(
512  ('b', Struct(
513  ('d', Scalar()),
514  )),
515  )
516 
517  Example 3
518  s1 = Struct(
519  ('a', Scalar()),
520  ('b', Struct(
521  ('d', Scalar()),
522  ))
523  )
524  s2 = Struct(
525  ('b', Struct(
526  ('c', Scalar())
527  ('d', Scalar())
528  )),
529  )
530  s1 - s2 == Struct(
531  ('a', Scalar()),
532  )
533  """
534  if not isinstance(other, Struct):
535  return NotImplemented
536 
537  children = OrderedDict(self.get_children())
538  for name, right_field in other.get_children():
539  if name in children:
540  left_field = children[name]
541  if type(left_field) == type(right_field):
542  if isinstance(left_field, Struct):
543  child = left_field - right_field
544  if child.get_children():
545  children[name] = child
546  continue
547  children.pop(name)
548  else:
549  raise TypeError(
550  "Type of left_field, " + str(type(left_field)) +
551  ", is not the same as that of right_field, " +
552  str(type(right_field)) +
553  ", yet they have the same field name, " + name)
554  return Struct(*(children.items()))

557 class Scalar(Field):
558  """Represents a typed scalar or tensor of fixed shape.
559 
560  A Scalar is a leaf in a schema tree, translating to exactly one tensor in
561  the dataset's underlying storage.
562 
563  Usually, the tensor storing the actual values of this field is a 1D tensor,
564  representing a series of values in its domain. It is possible however to
565  have higher rank values stored as a Scalar, as long as all entries have
566  the same shape.
567 
568  E.g.:
569 
570  Scalar(np.float64)
571 
572  Scalar field of type float32. Caffe2 will expect readers and
573  datasets to expose it as a 1D tensor of doubles (vector), where
574  the size of the vector is determined by this fields' domain.
575 
576  Scalar((np.int32, 5))
577 
578  Tensor field of type int32. Caffe2 will expect readers and
579  datasets to implement it as a 2D tensor (matrix) of shape (L, 5),
580  where L is determined by this fields' domain.
581 
582  Scalar((str, (10, 20)))
583 
584  Tensor field of type str. Caffe2 will expect readers and
585  datasets to implement it as a 3D tensor of shape (L, 10, 20),
586  where L is determined by this fields' domain.
587 
588  If the field type is unknown at construction time, call Scalar(), that will
589  default to np.void as its dtype.
590 
591  It is an error to pass a structured dtype to Scalar, since it would contain
592  more than one field. Instead, use from_dtype, which will construct
593  a nested `Struct` field reflecting the given dtype's structure.
594 
595  A Scalar can also contain a blob, which represents the value of this
596  Scalar. A blob can be either a numpy.ndarray, in which case it contain the
597  actual contents of the Scalar, or a BlobReference, which represents a
598  blob living in a caffe2 Workspace. If blob of different types are passed,
599  a conversion to numpy.ndarray is attempted.
600  """

672  def set(self, dtype=None, blob=None, metadata=None, unsafe=False):
673  """Set the type and/or blob of this scalar. See __init__ for details.
674 
675  Args:
676  dtype: can be any numpy type. If not provided and `blob` is
677  provided, it will be inferred. If no argument is provided,
678  this Scalar will be of type np.void.
679  blob: if provided, can be either a BlobReference or a
680  numpy.ndarray. If a value of different type is passed,
681  a conversion to numpy.ndarray is attempted. Strings aren't
682  accepted, since they can be ambiguous. If you want to pass
683  a string, to either BlobReference(blob) or np.array(blob).
684  metadata: optional instance of Metadata, if provided overrides
685  the metadata information of the scalar
686  """
687  if not unsafe:
688  logger.warning(
689  "Scalar should be considered immutable. Only call Scalar.set() "
690  "on newly created Scalar with unsafe=True. This will become an "
691  "error soon."
692  )
693  if blob is not None and isinstance(blob, basestring):
694  raise ValueError(
695  'Passing str blob to Scalar.set() is ambiguous. '
696  'Do either set(blob=np.array(blob)) or '
697  'set(blob=BlobReference(blob))'
698  )
699 
700  self._original_dtype = dtype
701  if dtype is not None:
702  dtype = np.dtype(dtype)
703  # If blob is not None and it is not a BlobReference, we assume that
704  # it is actual tensor data, so we will try to cast it to a numpy array.
705  if blob is not None and not isinstance(blob, BlobReference):
706  preserve_shape = isinstance(blob, np.ndarray)
707  if dtype is not None and dtype != np.void:
708  blob = np.array(blob, dtype=dtype.base)
709  # if array is empty we may need to reshape a little
710  if blob.size == 0 and not preserve_shape:
711  blob = blob.reshape((0, ) + dtype.shape)

797 def from_dtype(dtype, _outer_shape=()):
798  """Constructs a Caffe2 schema from the given numpy's dtype.
799 
800  Numpy supports scalar, array-like and structured datatypes, as long as
801  all the shapes are fixed. This function breaks down the given dtype into
802  a Caffe2 schema containing `Struct` and `Scalar` types.
803 
804  Fields containing byte offsets are not currently supported.
805  """
806  if not isinstance(dtype, np.dtype):
807  # wrap into a ndtype
808  shape = _outer_shape
809  dtype = np.dtype((dtype, _outer_shape))
810  else:
811  # concatenate shapes if necessary
812  shape = _outer_shape + dtype.shape
813  if shape != dtype.shape:
814  dtype = np.dtype((dtype.base, shape))
815 
816  if not dtype.fields:
817  return Scalar(dtype)
818 
819  struct_fields = []
820  for name, (fdtype, offset) in dtype.fields:
821  assert offset == 0, ('Fields with byte offsets are not supported.')
822  struct_fields += (name, from_dtype(fdtype, _outer_shape=shape))
823  return Struct(*struct_fields)
