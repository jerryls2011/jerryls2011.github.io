136  self._initializer = op
137  if op is not None:

148  def _infer_shape_from_initializer(self):
149  for arg in self.initializer.arg:
150  if arg.name == 'shape':
151  return list(arg.ints)
152  with workspace.WorkspaceGuard("model_init_by_loading_params"):
153  try:
154  net = core.Net("shape_checker")
155  net._net.op.extend([self.initializer])
156  shape_blob = net.NextScopedBlob(self.parameter + "_shape")

159  return workspace.FetchBlob(shape_blob).tolist()

184  def __init__(self, model, prefix, input_record,
185  predict_input_record_fields=None, tags=None, **kwargs):
186  """
187  Base class for model layers. Layer is an abstraction that allows to
188  provide model description in terms of meta-operators, where each of the
189  meta-operators can have different implementations for training,
190  evaluation and prediction, that are instantiated later. As an example
191  SampledSoftmax can do something related to sampling depending on
192  supervision during the training and just apply softmax if it's used for
193  prediction/evaluation.
194 
195  All inputs/outputs from layers are represented as a record (instance of
196  schema bounded to blobs) and are accessible through input_record and
197  output_schema. If Layer needs to have only a subset of inputs/provides
198  subset of outputs during the inference - it should provide
199  predict_input_record and predict_output_schema correspondingly (those
200  records are expected to be a subset of input_record/output_schema).
201 
202  Each layer has a list of Tags associated with it, that depends on
203  current context and arguments. It's possible to use those tags during
204  the instantiation time.
205 
206  """
207  self.name = model.next_layer_name(prefix)
208  self.model = model
209  self.kwargs = kwargs
210  self._input_record = input_record
211  if predict_input_record_fields:
212  if not isinstance(predict_input_record_fields, list):
213  predict_input_record_fields = [predict_input_record_fields]
214  self._predict_input_record = self._input_record[
215  predict_input_record_fields]
216  else:
217  self._predict_input_record = None
218 
219  self.request_only = True
220  if len(input_record.all_scalars()) == 0:
221  self.request_only = False
222  for scalar in input_record.all_scalars():
223  if not is_request_only_scalar(scalar):
224  self.request_only = False
225  break
