22 def generate_data(T, shape, num_labels, fixed_shape):
23  '''
24  Fill a queue with input data
25  '''
26  log.info("Generating T={} sequence batches".format(T))
27 
28  generate_input_init_net = core.Net('generate_input_init')
29  queue = generate_input_init_net.CreateBlobsQueue(
30  [], "inputqueue", num_blobs=1, capacity=T,
31  )
32  label_queue = generate_input_init_net.CreateBlobsQueue(
33  [], "labelqueue", num_blobs=1, capacity=T,
34  )
35 
36  workspace.RunNetOnce(generate_input_init_net)
37  generate_input_net = core.Net('generate_input')
38 
39  generate_input_net.EnqueueBlobs([queue, "scratch"], ["scratch"])
40  generate_input_net.EnqueueBlobs([label_queue, "label_scr"], ["label_scr"])
41  np.random.seed(2603)
42 
43  entry_counts = []
44  for t in range(T):
45  if (t % (max(10, T // 10)) == 0):
46  print("Generating data {}/{}".format(t, T))
47  # Randomize the seqlength
48  random_shape = (
49  [np.random.randint(1, shape[0])] + shape[1:]
50  if t > 0 and not fixed_shape else shape
51  )
52  X = np.random.rand(*random_shape).astype(np.float32)
53  batch_size = random_shape[1]
54  L = num_labels * batch_size
55  labels = (np.random.rand(random_shape[0]) * L).astype(np.int32)
56  workspace.FeedBlob("scratch", X)
57  workspace.FeedBlob("label_scr", labels)
58  workspace.RunNetOnce(generate_input_net.Proto())
59  entry_counts.append(random_shape[0] * random_shape[1])
60 
61  log.info("Finished data generation")
62 
63  return queue, label_queue, entry_counts

66 def create_model(args, queue, label_queue, input_shape):
67  model = model_helper.ModelHelper(name="LSTM_bench")
68  seq_lengths, target = \
69  model.net.AddExternalInputs(
70  'seq_lengths',
71  'target',
72  )
73 
74  input_blob = model.net.DequeueBlobs(queue, "input_data")
75  labels = model.net.DequeueBlobs(label_queue, "label")
76 
77  init_blobs = []
78  if args.implementation in ["own", "static", "static_dag"]:
79  T = None
80  if "static" in args.implementation:
81  assert args.fixed_shape, \
82  "Random input length is not static RNN compatible"
83  T = args.seq_length
84  print("Using static RNN of size {}".format(T))
85 
86  for i in range(args.num_layers):
87  hidden_init, cell_init = model.net.AddExternalInputs(
88  "hidden_init_{}".format(i),
89  "cell_init_{}".format(i)
90  )
91  init_blobs.extend([hidden_init, cell_init])
92 
93  output, last_hidden, _, last_state = rnn_cell.LSTM(
94  model=model,
95  input_blob=input_blob,
96  seq_lengths=seq_lengths,
97  initial_states=init_blobs,
98  dim_in=args.input_dim,
99  dim_out=[args.hidden_dim] * args.num_layers,
100  scope="lstm1",
101  memory_optimization=args.memory_optimization,
102  forward_only=args.forward_only,
103  drop_states=True,
104  return_last_layer_only=True,
105  static_rnn_unroll_size=T,
106  )
107 
108  if "dag" in args.implementation:
109  print("Using DAG net type")
110  model.net.Proto().type = 'dag'
111  model.net.Proto().num_workers = 4
112  elif args.implementation == "cudnn":
113  # We need to feed a placeholder input so that RecurrentInitOp
114  # can infer the dimensions.
115  init_blobs = model.net.AddExternalInputs("hidden_init", "cell_init")
116  model.param_init_net.ConstantFill([], input_blob, shape=input_shape)
117  output, last_hidden, _ = rnn_cell.cudnn_LSTM(
118  model=model,
119  input_blob=input_blob,
120  initial_states=init_blobs,
121  dim_in=args.input_dim,
122  dim_out=args.hidden_dim,
123  scope="cudnnlstm",
124  num_layers=args.num_layers,
125  )
126 
127  else:
128  assert False, "Unknown implementation"
129 
130  weights = model.net.UniformFill(labels, "weights")
131  softmax, loss = model.net.SoftmaxWithLoss(
132  [model.Flatten(output), labels, weights],
133  ['softmax', 'loss'],
134  )
135 
136  if not args.forward_only:
137  model.AddGradientOperators([loss])
138 
139  # carry states over
140  for init_blob in init_blobs:
141  model.net.Copy(last_hidden, init_blob)
142 
143  sz = args.hidden_dim
144  if args.implementation == "cudnn":
145  sz *= args.num_layers
146  workspace.FeedBlob(init_blob, np.zeros(
147  [1, args.batch_size, sz], dtype=np.float32
148  ))
149  return model, output

152 def Caffe2LSTM(args):
153  T = args.data_size // args.batch_size
154 
155  input_blob_shape = [args.seq_length, args.batch_size, args.input_dim]
156  queue, label_queue, entry_counts = generate_data(T // args.seq_length,
157  input_blob_shape,
158  args.hidden_dim,
159  args.fixed_shape)
160 
161  workspace.FeedBlob(
162  "seq_lengths",
163  np.array([args.seq_length] * args.batch_size, dtype=np.int32)
164  )
165 
166  model, output = create_model(args, queue, label_queue, input_blob_shape)
167 
168  workspace.RunNetOnce(model.param_init_net)
169  workspace.CreateNet(model.net)
170 
171  start_time = time.time()
172  num_iters = T // args.seq_length
173  total_iters = 0
174 
175  # Run the Benchmark
176  log.info("------ Warming up ------")
177  workspace.RunNet(model.net.Proto().name)
178 
179  if (args.gpu):
180  log.info("Memory stats:")
181  stats = utils.GetGPUMemoryUsageStats()
182  log.info("GPU memory:\t{} MB".format(stats['max_total'] / 1024 / 1024))
183 
184  log.info("------ Starting benchmark ------")
185  start_time = time.time()
186  last_time = time.time()
187  for iteration in range(1, num_iters, args.iters_to_report):
188  iters_once = min(args.iters_to_report, num_iters - iteration)
189  total_iters += iters_once
190  workspace.RunNet(model.net.Proto().name, iters_once)
191 
192  new_time = time.time()
193  log.info(
194  "Iter: {} / {}. Entries Per Second: {}k.".format(
195  iteration,
196  num_iters,
197  np.sum(entry_counts[iteration:iteration + iters_once]) /
198  (new_time - last_time) // 100 / 10,
199  )
200  )
201  last_time = new_time
202 
203  log.info("Done. Total EPS excluding 1st iteration: {}k".format(
204  np.sum(entry_counts[1:]) / (time.time() - start_time) // 100 / 10,
205  ))
206 
207  if (args.gpu):
208  log.info("Memory stats:")
209  stats = utils.GetGPUMemoryUsageStats()
210  log.info("GPU memory:\t{} MB".format(stats['max_total'] / 1024 / 1024))
211  if (stats['max_total'] != stats['total']):
212  log.warning(
213  "Max usage differs from current total usage: {} > {}".
214  format(stats['max_total'], stats['total'])
215  )
216  log.warning("This means that costly deallocations occured.")
217 
218  return time.time() - start_time
