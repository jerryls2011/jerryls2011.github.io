192  # global_init_net in meta_net_def will load parameters from
193  # predictor_constants.PREDICTOR_DBREADER
194  create_db = core.CreateOperator(

198  assert workspace.RunOperatorOnce(create_db), (
199  'Failed to create db {}'.format(filename))
200 
201  # predictor_constants.META_NET_DEF is always stored before the parameters
202  load_meta_net_def = core.CreateOperator(
203  'Load',
204  [core.BlobReference(predictor_constants.PREDICTOR_DBREADER)],
205  [core.BlobReference(predictor_constants.META_NET_DEF)])
206  assert workspace.RunOperatorOnce(load_meta_net_def)
207

209  meta_net_def = serde.deserialize_protobuf_struct(
210  blob if isinstance(blob, bytes)

214  if device_option is None:
