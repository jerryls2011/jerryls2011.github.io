30 class RNNCell(object):
31  '''
32  Base class for writing recurrent / stateful operations.
33 
34  One needs to implement 3 methods: _apply, prepare_input and get_state_names.
35  As a result base class will provice apply_over_sequence method, which
36  allows you to apply recurrent operations over a sequence of any length.
37  '''
38  def __init__(self, name, forward_only=False):
39  self.name = name
40  self.recompute_blobs = []
41  self.forward_only = forward_only
42 
43  def scope(self, name):
44  return self.name + '/' + name if self.name is not None else name
45 
46  def apply_over_sequence(
47  self,
48  model,
49  inputs,
50  seq_lengths,
51  initial_states,
52  outputs_with_grads=None,
53  ):
54  preprocessed_inputs = self.prepare_input(model, inputs)
55  step_model = ModelHelper(name=self.name, param_model=model)
56  input_t, timestep = step_model.net.AddScopedExternalInputs(
57  'input_t',
58  'timestep',
59  )
60  states_prev = step_model.net.AddScopedExternalInputs(*[
61  s + '_prev' for s in self.get_state_names()
62  ])
63  states = self._apply(
64  model=step_model,
65  input_t=input_t,
66  seq_lengths=seq_lengths,
67  states=states_prev,
68  timestep=timestep,
69  )
70 
71  if outputs_with_grads is None:
72  outputs_with_grads = [self.get_output_state_index() * 2]
73 
74  # states_for_all_steps consists of combination of
75  # states gather for all steps and final states. It looks like this:
76  # (state_1_all, state_1_final, state_2_all, state_2_final, ...)
77  states_for_all_steps = recurrent.recurrent_net(
78  net=model.net,
79  cell_net=step_model.net,
80  inputs=[(input_t, preprocessed_inputs)],
81  initial_cell_inputs=list(zip(states_prev, initial_states)),
82  links=dict(zip(states_prev, states)),
83  timestep=timestep,
84  scope=self.name,
85  outputs_with_grads=outputs_with_grads,
86  recompute_blobs_on_backward=self.recompute_blobs,
87  )
88 
89  output = self._prepare_output_sequence(
90  model,
91  states_for_all_steps,
92  )
93  return output, states_for_all_steps
94 
95  def apply(self, model, input_t, seq_lengths, states, timestep):
96  input_t = self.prepare_input(model, input_t)
97  states = self._apply(
98  model, input_t, seq_lengths, states, timestep)
99  output = self._prepare_output(model, states)
100  return output, states

102  def _apply(
103  self,
104  model,
105  input_t,
106  seq_lengths,
107  states,
108  timestep,
109  extra_inputs,
110  ):
111  '''
112  A single step of a recurrent network.
113 
114  model: ModelHelper object new operators would be added to
115 
116  input_t: single input with shape (1, batch_size, input_dim)
117 
118  seq_lengths: blob containing sequence lengths which would be passed to
119  LSTMUnit operator
120 
121  states: previous recurrent states
122 
123  timestep: current recurrent iteration. Could be used together with
124  seq_lengths in order to determine, if some shorter sequences
125  in the batch have already ended.
126 
127  extra_inputs: list of tuples (input, dim). specifies additional input
128  which is not subject to prepare_input(). (useful when a cell is a
129  component of a larger recurrent structure, e.g., attention)
130  '''
131  raise NotImplementedError('Abstract method')
132 
133  def prepare_input(self, model, input_blob):
134  '''
135  If some operations in _apply method depend only on the input,
136  not on recurrent states, they could be computed in advance.
137 
138  model: ModelHelper object new operators would be added to
139 
140  input_blob: either the whole input sequence with shape
141  (sequence_length, batch_size, input_dim) or a single input with shape
142  (1, batch_size, input_dim).
143  '''
144  return input_blob

147  '''
148  Return index into state list of the "primary" step-wise output.

150  return 0

154  Return the names of the recurrent states.

158  raise NotImplementedError('Abstract method')

162  Specifies the dimension (number of units) of stepwise output.

166  def _prepare_output(self, model, states):
167  '''
168  Allows arbitrary post-processing of primary output.
169  '''
170  return states[self.get_output_state_index()]

172  def _prepare_output_sequence(self, model, state_outputs):
173  '''
174  Allows arbitrary post-processing of primary sequence output.
175

180  return state_outputs[output_sequence_index]

195  self.input_size = input_size

201  def _apply(
202  self,
203  model,
204  input_t,
205  seq_lengths,
206  states,
207  timestep,
208  extra_inputs=None,
209  ):
210  hidden_t_prev, cell_t_prev = states
211 
212  fc_input = hidden_t_prev
213  fc_input_dim = self.hidden_size
214 
215  if extra_inputs is not None:
216  extra_input_blobs, extra_input_sizes = zip(*extra_inputs)
217  fc_input, _ = model.net.Concat(
218  [hidden_t_prev] + list(extra_input_blobs),
219  [
220  self.scope('gates_concatenated_input_t'),
221  self.scope('_gates_concatenated_input_t_concat_dims'),
222  ],
223  axis=2,
224  )

226

228  model,
229  fc_input,
230  self.scope('gates_t'),
231  dim_in=fc_input_dim,
232  dim_out=4 * self.hidden_size,
233  axis=2,
234  )
235  model.net.Sum([gates_t, input_t], gates_t)
236 
237  hidden_t, cell_t = model.net.LSTMUnit(
238  [
239  hidden_t_prev,
240  cell_t_prev,
241  gates_t,
242  seq_lengths,
243  timestep,
244  ],
245  list(self.get_state_names()),
246  forget_bias=self.forget_bias,
247  drop_states=self.drop_states,
248  )
249  model.net.AddExternalOutputs(hidden_t, cell_t)
250  if self.memory_optimization:
251  self.recompute_blobs = [gates_t]
252  return hidden_t, cell_t

283 class MILSTMCell(LSTMCell):
284 
285  def _apply(
286  self,
287  model,
288  input_t,
289  seq_lengths,
290  states,
291  timestep,
292  extra_inputs=None,
293  ):
294  hidden_t_prev, cell_t_prev = states
295 
296  fc_input = hidden_t_prev
297  fc_input_dim = self.hidden_size
298 
299  if extra_inputs is not None:
300  extra_input_blobs, extra_input_sizes = zip(*extra_inputs)
301  fc_input, _ = model.net.Concat(
302  [hidden_t_prev] + list(extra_input_blobs),
303  [
304  self.scope('gates_concatenated_input_t'),
305  self.scope('_gates_concatenated_input_t_concat_dims'),
306  ],
307  axis=2,
308  )
309  fc_input_dim += sum(extra_input_sizes)
310 
311  prev_t = brew.fc(
312  model,
313  fc_input,
314  self.scope('prev_t'),
315  dim_in=fc_input_dim,
316  dim_out=4 * self.hidden_size,
317  axis=2,
318  )
319 
320  # defining initializers for MI parameters
321  alpha = model.create_param(
322  self.scope('alpha'),
323  shape=[4 * self.hidden_size],
324  initializer=Initializer('ConstantFill', value=1.0),
325  )
326  beta_h = model.create_param(
327  self.scope('beta1'),
328  shape=[4 * self.hidden_size],
329  initializer=Initializer('ConstantFill', value=1.0),
330  )
331  beta_i = model.create_param(
332  self.scope('beta2'),
333  shape=[4 * self.hidden_size],
334  initializer=Initializer('ConstantFill', value=1.0),
335  )
336  b = model.create_param(
337  self.scope('b'),
338  shape=[4 * self.hidden_size],
339  initializer=Initializer('ConstantFill', value=0.0),
340  )
341 
342  # alpha * input_t + beta_h
343  # Shape: [1, batch_size, 4 * hidden_size]
344  alpha_by_input_t_plus_beta_h = model.net.ElementwiseLinear(
345  [input_t, alpha, beta_h],
346  self.scope('alpha_by_input_t_plus_beta_h'),
347  axis=2,
348  )
349  # (alpha * input_t + beta_h) * prev_t =
350  # alpha * input_t * prev_t + beta_h * prev_t
351  # Shape: [1, batch_size, 4 * hidden_size]
352  alpha_by_input_t_plus_beta_h_by_prev_t = model.net.Mul(
353  [alpha_by_input_t_plus_beta_h, prev_t],
354  self.scope('alpha_by_input_t_plus_beta_h_by_prev_t')
355  )
356  # beta_i * input_t + b
357  # Shape: [1, batch_size, 4 * hidden_size]
358  beta_i_by_input_t_plus_b = model.net.ElementwiseLinear(
359  [input_t, beta_i, b],
360  self.scope('beta_i_by_input_t_plus_b'),
361  axis=2,
362  )
363  # alpha * input_t * prev_t + beta_h * prev_t + beta_i * input_t + b
364  # Shape: [1, batch_size, 4 * hidden_size]
365  gates_t = model.net.Sum(
366  [alpha_by_input_t_plus_beta_h_by_prev_t, beta_i_by_input_t_plus_b],
367  self.scope('gates_t')
368  )
369  hidden_t, cell_t = model.net.LSTMUnit(
370  [hidden_t_prev, cell_t_prev, gates_t, seq_lengths, timestep],
371  [self.scope('hidden_t_intermediate'), self.scope('cell_t')],
372  forget_bias=self.forget_bias,
373  drop_states=self.drop_states,
374  )
375  model.net.AddExternalOutputs(
376  cell_t,
377  hidden_t,
378  )
379  if self.memory_optimization:
380  self.recompute_blobs = [gates_t]
381  return hidden_t, cell_t

453 class MultiRNNCell(RNNCell):
454  '''
455  Multilayer RNN via the composition of RNNCell instance.
456 
457  It is the resposibility of calling code to ensure the compatibility
458  of the successive layers in terms of input/output dimensiality, etc.,
459  and to ensure that their blobs do not have name conflicts, typically by
460  creating the cells with names that specify layer number.
461 
462  Assumes first state (recurrent output) for each layer should be the input
463  to the next layer.
464  '''
465 
466  def __init__(self, cells, residual_output_layers=None, **kwargs):
467  '''
468  cells: list of RNNCell instances, from input to output side.
469 
470  name: string designating network component (for scoping)
471 
472  residual_output_layers: list of indices of layers whose input will
473  be added elementwise to their output elementwise. (It is the
474  responsibility of the client code to ensure shape compatibility.)
475  Note that layer 0 (zero) cannot have residual output because of the
476  timing of prepare_input().
477 
478  forward_only: used to construct inference-only network.
479  '''
480  super(MultiRNNCell, self).__init__(**kwargs)
481  self.cells = cells
482 
483  if residual_output_layers is None:
484  self.residual_output_layers = []
485  else:
486  self.residual_output_layers = residual_output_layers
487 
488  output_index_per_layer = []
489  base_index = 0
490  for cell in self.cells:
491  output_index_per_layer.append(
492  base_index + cell.get_output_state_index(),
493  )
494  base_index += len(cell.get_state_names())
495 
496  self.output_connected_layers = []
497  self.output_indices = []
498  for i in range(len(self.cells) - 1):
499  if (i + 1) in self.residual_output_layers:
500  self.output_connected_layers.append(i)
501  self.output_indices.append(output_index_per_layer[i])
502  else:
503  self.output_connected_layers = []
504  self.output_indices = []
505  self.output_connected_layers.append(len(self.cells) - 1)
506  self.output_indices.append(output_index_per_layer[-1])
507 
508  self.state_names = []
509  for cell in self.cells:
510  self.state_names.extend(cell.get_state_names())
511 
512  if len(self.state_names) != len(set(self.state_names)):
513  duplicates = {
514  state_name for state_name in self.state_names
515  if self.state_names.count(state_name) > 1
516  }
517  raise RuntimeError(
518  'Duplicate state names in MultiRNNCell: {}'.format(
519  list(duplicates),
520  ),
521  )
522 
523  def prepare_input(self, model, input_blob):
524  return self.cells[0].prepare_input(model, input_blob)

526  def _apply(
527  self,
528  model,
529  input_t,
530  seq_lengths,
531  states,
532  timestep,
533  extra_inputs=None,
534  ):
535 
536  states_per_layer = [len(cell.get_state_names()) for cell in self.cells]
537  assert len(states) == sum(states_per_layer)
538 
539  next_states = []
540  states_index = 0
541 
542  layer_input = input_t
543  for i, layer_cell in enumerate(self.cells):
544  num_states = states_per_layer[i]
545  layer_states = states[states_index:(states_index + num_states)]
546  states_index += num_states
547 
548  if i > 0:
549  prepared_input = layer_cell.prepare_input(model, layer_input)
550  else:
551  prepared_input = layer_input
552 
553  layer_next_states = layer_cell._apply(
554  model,
555  prepared_input,
556  seq_lengths,
557  layer_states,
558  timestep,
559  extra_inputs=(None if i > 0 else extra_inputs),
560  )
561  # Since we're using here non-public method _apply, instead of apply,
562  # we have to manually extract output from states
563  if i != len(self.cells) - 1:
564  layer_output = layer_cell._prepare_output(
565  model,
566  layer_next_states,
567  )
568  if i > 0 and i in self.residual_output_layers:
569  layer_input = model.net.Sum(
570  [layer_output, layer_input],
571  self.scope('residual_output_{}'.format(i)),
572  )
573  else:
574  layer_input = layer_output
575 
576  next_states.extend(layer_next_states)
577  return next_states

579  def get_state_names(self):
580  return self.state_names
581 
582  def get_output_state_index(self):
583  index = 0
584  for cell in self.cells[:-1]:
585  index += len(cell.get_state_names())
586  index += self.cells[-1].get_output_state_index()
587  return index

589  def _prepare_output(self, model, states):
590  connected_outputs = []
591  state_index = 0
592  for i, cell in enumerate(self.cells):
593  num_states = len(cell.get_state_names())
594  if i in self.output_connected_layers:
595  layer_states = states[state_index:state_index + num_states]
596  layer_output = cell._prepare_output(
597  model,
598  layer_states
599  )
600  connected_outputs.append(layer_output)
601  state_index += num_states
602 
603  output = model.net.Sum(
604  connected_outputs,
605  self.scope('residual_output'),
606  )
607  return output
608 
609  def _prepare_output_sequence(self, model, states):
610  connected_outputs = []
611  state_index = 0
612  for i, cell in enumerate(self.cells):
613  num_states = 2 * len(cell.get_state_names())
614  if i in self.output_connected_layers:
615  layer_states = states[state_index:state_index + num_states]
616  layer_output = cell._prepare_output_sequence(
617  model,
618  layer_states
619  )
620  connected_outputs.append(layer_output)
621  state_index += num_states
622 
623  output = model.net.Sum(
624  connected_outputs,
625  self.scope('residual_output_sequence'),
626  )
627  return output

909 def _LSTM(
910  cell_class,
911  model,
912  input_blob,
913  seq_lengths,
914  initial_states,
915  dim_in,
916  dim_out,
917  scope,
918  outputs_with_grads=(0,),
919  return_params=False,
920  memory_optimization=False,
921  forget_bias=0.0,
922  forward_only=False,
923  drop_states=False,
924  return_last_layer_only=True,
925  static_rnn_unroll_size=None,
926 ):
927  '''
928  Adds a standard LSTM recurrent network operator to a model.
929 
930  cell_class: LSTMCell or compatible subclass
931 
932  model: ModelHelper object new operators would be added to
933 
934  input_blob: the input sequence in a format T x N x D
935  where T is sequence size, N - batch size and D - input dimension
936 
937  seq_lengths: blob containing sequence lengths which would be passed to
938  LSTMUnit operator
939 
940  initial_states: a list of (2 * num_layers) blobs representing the initial
941  hidden and cell states of each layer. If this argument is None,
942  these states will be added to the model as network parameters.
943 
944  dim_in: input dimension
945 
946  dim_out: number of units per LSTM layer
947  (use int for single-layer LSTM, list of ints for multi-layer)
948 
949  outputs_with_grads : position indices of output blobs for LAST LAYER which
950  will receive external error gradient during backpropagation.
951  These outputs are: (h_all, h_last, c_all, c_last)
952 
953  return_params: if True, will return a dictionary of parameters of the LSTM
954 
955  memory_optimization: if enabled, the LSTM step is recomputed on backward
956  step so that we don't need to store forward activations for each
957  timestep. Saves memory with cost of computation.
958 
959  forget_bias: forget gate bias (default 0.0)
960 
961  forward_only: whether to create a backward pass
962 
963  drop_states: drop invalid states, passed through to LSTMUnit operator
964 
965  return_last_layer_only: only return outputs from final layer
966  (so that length of results does depend on number of layers)
967 
968  static_rnn_unroll_size: if not None, we will use static RNN which is
969  unrolled into Caffe2 graph. The size of the unroll is the value of
970  this parameter.
971  '''
972  if type(dim_out) is not list and type(dim_out) is not tuple:
973  dim_out = [dim_out]
974  num_layers = len(dim_out)
975 
976  cells = []
977  for i in range(num_layers):
978  name = '{}/layer_{}'.format(scope, i) if num_layers > 1 else scope
979  cell = cell_class(
980  input_size=(dim_in if i == 0 else dim_out[i - 1]),
981  hidden_size=dim_out[i],
982  forget_bias=forget_bias,
983  memory_optimization=memory_optimization,
984  name=name,
985  forward_only=forward_only,
986  drop_states=drop_states,
987  )
988  cells.append(cell)
989 
990  cell = MultiRNNCell(
991  cells,
992  name=scope,
993  forward_only=forward_only,
994  ) if num_layers > 1 else cells[0]
995 
996  cell = (
997  cell if static_rnn_unroll_size is None
998  else UnrolledCell(cell, static_rnn_unroll_size))
999 
1000  if initial_states is None:
1001  initial_states = []
1002  for i in range(num_layers):
1003  with core.NameScope(scope):
1004  suffix = '_{}'.format(i) if num_layers > 1 else ''
1005  initial_hidden = model.create_param(
1006  'initial_hidden_state' + suffix,
1007  shape=[dim_out[i]],
1008  initializer=Initializer('ConstantFill', value=0.0),
1009  )
1010  initial_cell = model.create_param(
1011  'initial_cell_state' + suffix,
1012  shape=[dim_out[i]],
1013  initializer=Initializer('ConstantFill', value=0.0),
1014  )
1015  initial_states.extend([initial_hidden, initial_cell])
1016 
1017  assert len(initial_states) == 2 * num_layers, \
1018  "Incorrect initial_states, was expecting 2 * num_layers elements" \
1019  + " but had only {}".format(len(initial_states))
1020 
1021  # outputs_with_grads argument indexes into final layer
1022  outputs_with_grads = [4 * (num_layers - 1) + i for i in outputs_with_grads]
1023  _, result = cell.apply_over_sequence(
1024  model=model,
1025  inputs=input_blob,
1026  seq_lengths=seq_lengths,
1027  initial_states=initial_states,
1028  outputs_with_grads=outputs_with_grads,
1029  )
1030 
1031  if return_last_layer_only:
1032  result = result[4 * (num_layers - 1):]
1033  if return_params:
1034  result = list(result) + [{
1035  'input': cell.get_input_params(),
1036  'recurrent': cell.get_recurrent_params(),
1037  }]
1038  return tuple(result)

1050  def apply_over_sequence(
1051  self,
1052  model,
1053  inputs,
1054  seq_lengths,
1055  initial_states,
1056  outputs_with_grads=None,
1057  ):
1058  inputs = self.cell.prepare_input(model, inputs)
1059 
1060  # Now they are blob references - outputs of splitting the input sequence
1061  split_inputs = model.net.Split(
1062  inputs,
1063  [str(inputs) + "_timestep_{}".format(i)
1064  for i in range(self.T)],
1065  axis=0)
1066  if self.T == 1:
1067  split_inputs = [split_inputs]
1068 
1069  states = initial_states
1070  all_states = []
1071  for t in range(0, self.T):
1072  scope_name = "timestep_{}".format(t)
1073  # Parameters of all timesteps are shared
1074  with ParameterSharing({scope_name: ''}),\
1075  scope.NameScope(scope_name):
1076  timestep = model.param_init_net.ConstantFill(
1077  [], "timestep", value=t, shape=[1],
1078  dtype=core.DataType.INT32,
1079  device_option=core.DeviceOption(caffe2_pb2.CPU))
1080  states = self.cell._apply(
1081  model=model,
1082  input_t=split_inputs[t],
1083  seq_lengths=seq_lengths,
1084  states=states,
1085  timestep=timestep,
1086  )
1087  all_states.append(states)
1088 
1089  all_states = zip(*all_states)
1090  all_states = [
1091  model.net.Concat(
1092  list(full_output),
1093  [
1094  str(full_output[0])[len("timestep_0/"):] + "_concat",
1095  str(full_output[0])[len("timestep_0/"):] + "_concat_info"
1096 
1097  ],
1098  axis=0)[0]
1099  for full_output in all_states
1100  ]
1101  outputs = tuple(
1102  six.next(it) for it in
1103  itertools.cycle([iter(all_states), iter(states)])
1104  )
1105  outputs_without_grad = set(range(len(outputs))) - set(
1106  outputs_with_grads)
1107  for i in outputs_without_grad:
1108  model.net.ZeroGradient(outputs[i], [])
1109  logging.debug("Added 0 gradients for blobs:",
1110  [outputs[i] for i in outputs_without_grad])
1111 
1112  final_output = self.cell._prepare_output_sequence(model, outputs)
1113 
1114  return final_output, outputs

1158 def cudnn_LSTM(model, input_blob, initial_states, dim_in, dim_out,
1159  scope, recurrent_params=None, input_params=None,
1160  num_layers=1, return_params=False):
1161  '''
1162  CuDNN version of LSTM for GPUs.
1163  input_blob Blob containing the input. Will need to be available
1164  when param_init_net is run, because the sequence lengths
1165  and batch sizes will be inferred from the size of this
1166  blob.
1167  initial_states tuple of (hidden_init, cell_init) blobs
1168  dim_in input dimensions
1169  dim_out output/hidden dimension
1170  scope namescope to apply
1171  recurrent_params dict of blobs containing values for recurrent
1172  gate weights, biases (if None, use random init values)
1173  See GetLSTMParamNames() for format.
1174  input_params dict of blobs containing values for input
1175  gate weights, biases (if None, use random init values)
1176  See GetLSTMParamNames() for format.
1177  num_layers number of LSTM layers
1178  return_params if True, returns (param_extract_net, param_mapping)
1179  where param_extract_net is a net that when run, will
1180  populate the blobs specified in param_mapping with the
1181  current gate weights and biases (input/recurrent).
1182  Useful for assigning the values back to non-cuDNN
1183  LSTM.
1184  '''
1185  with core.NameScope(scope):
1186  weight_params = GetLSTMParamNames()['weights']
1187  bias_params = GetLSTMParamNames()['biases']
1188 
1189  input_weight_size = dim_out * dim_in
1190  upper_layer_input_weight_size = dim_out * dim_out
1191  recurrent_weight_size = dim_out * dim_out
1192  input_bias_size = dim_out
1193  recurrent_bias_size = dim_out
1194 
1195  def init(layer, pname, input_type):
1196  input_weight_size_for_layer = input_weight_size if layer == 0 else \
1197  upper_layer_input_weight_size
1198  if pname in weight_params:
1199  sz = input_weight_size_for_layer if input_type == 'input' \
1200  else recurrent_weight_size
1201  elif pname in bias_params:
1202  sz = input_bias_size if input_type == 'input' \
1203  else recurrent_bias_size
1204  else:
1205  assert False, "unknown parameter type {}".format(pname)
1206  return model.param_init_net.UniformFill(
1207  [],
1208  "lstm_init_{}_{}_{}".format(input_type, pname, layer),
1209  shape=[sz])
1210 
1211  # Multiply by 4 since we have 4 gates per LSTM unit
1212  first_layer_sz = input_weight_size + recurrent_weight_size + \
1213  input_bias_size + recurrent_bias_size
1214  upper_layer_sz = upper_layer_input_weight_size + \
1215  recurrent_weight_size + input_bias_size + \
1216  recurrent_bias_size
1217  total_sz = 4 * (first_layer_sz + (num_layers - 1) * upper_layer_sz)
1218 
1219  weights = model.create_param(
1220  'lstm_weight',
1221  shape=[total_sz],
1222  initializer=Initializer('UniformFill'),
1223  tags=ParameterTags.WEIGHT,
1224  )
1225 
1226  lstm_args = {
1227  'hidden_size': dim_out,
1228  'rnn_mode': 'lstm',
1229  'bidirectional': 0, # TODO
1230  'dropout': 1.0, # TODO
1231  'input_mode': 'linear', # TODO
1232  'num_layers': num_layers,
1233  'engine': 'CUDNN'
1234  }
1235 
1236  param_extract_net = core.Net("lstm_param_extractor")
1237  param_extract_net.AddExternalInputs([input_blob, weights])
1238  param_extract_mapping = {}
1239 
1240  # Populate the weights-blob from blobs containing parameters for
1241  # the individual components of the LSTM, such as forget/input gate
1242  # weights and bises. Also, create a special param_extract_net that
1243  # can be used to grab those individual params from the black-box
1244  # weights blob. These results can be then fed to InitFromLSTMParams()
1245  for input_type in ['input', 'recurrent']:
1246  param_extract_mapping[input_type] = {}
1247  p = recurrent_params if input_type == 'recurrent' else input_params
1248  if p is None:
1249  p = {}
1250  for pname in weight_params + bias_params:
1251  for j in range(0, num_layers):
1252  values = p[pname] if pname in p else init(j, pname, input_type)
1253  model.param_init_net.RecurrentParamSet(
1254  [input_blob, weights, values],
1255  weights,
1256  layer=j,
1257  input_type=input_type,
1258  param_type=pname,
1259  **lstm_args
1260  )
1261  if pname not in param_extract_mapping[input_type]:
1262  param_extract_mapping[input_type][pname] = {}
1263  b = param_extract_net.RecurrentParamGet(
1264  [input_blob, weights],
1265  ["lstm_{}_{}_{}".format(input_type, pname, j)],
1266  layer=j,
1267  input_type=input_type,
1268  param_type=pname,
1269  **lstm_args
1270  )
1271  param_extract_mapping[input_type][pname][j] = b
1272 
1273  (hidden_input_blob, cell_input_blob) = initial_states
1274  output, hidden_output, cell_output, rnn_scratch, dropout_states = \
1275  model.net.Recurrent(
1276  [input_blob, cell_input_blob, cell_input_blob, weights],
1277  ["lstm_output", "lstm_hidden_output", "lstm_cell_output",
1278  "lstm_rnn_scratch", "lstm_dropout_states"],
1279  seed=random.randint(0, 100000), # TODO: dropout seed
1280  **lstm_args
1281  )
1282  model.net.AddExternalOutputs(
1283  hidden_output, cell_output, rnn_scratch, dropout_states)
1284 
1285  if return_params:
1286  param_extract = param_extract_net, param_extract_mapping
1287  return output, hidden_output, cell_output, param_extract
1288  else:
1289  return output, hidden_output, cell_output

1293  model,
1294  decoder_inputs,
1295  decoder_input_lengths,
1296  initial_decoder_hidden_state,
1297  initial_decoder_cell_state,
1298  initial_attention_weighted_encoder_context,
1299  encoder_output_dim,

1301  encoder_lengths,
1302  decoder_input_dim,
1303  decoder_state_dim,
1304  scope,
1305  attention_type=AttentionType.Regular,
1306  outputs_with_grads=(0, 4),
1307  weighted_encoder_outputs=None,
1308  lstm_memory_optimization=False,
1309  attention_memory_optimization=False,
1310  forget_bias=0.0,
1311  forward_only=False,
1312 ):
1313  '''
1314  Adds a LSTM with attention mechanism to a model.
1315 
1316  The implementation is based on https://arxiv.org/abs/1409.0473, with
1317  a small difference in the order
1318  how we compute new attention context and new hidden state, similarly to
1319  https://arxiv.org/abs/1508.04025.
1320 
1321  The model uses encoder-decoder naming conventions,
1322  where the decoder is the sequence the op is iterating over,
1323  while computing the attention context over the encoder.
1324 
1325  model: ModelHelper object new operators would be added to
1326 
1327  decoder_inputs: the input sequence in a format T x N x D
1328  where T is sequence size, N - batch size and D - input dimension
1329 
1330  decoder_input_lengths: blob containing sequence lengths
1331  which would be passed to LSTMUnit operator
1332 
1333  initial_decoder_hidden_state: initial hidden state of LSTM
1334 
1335  initial_decoder_cell_state: initial cell state of LSTM
1336 
1337  initial_attention_weighted_encoder_context: initial attention context
1338 
1339  encoder_output_dim: dimension of encoder outputs
1340 
1341  encoder_outputs: the sequence, on which we compute the attention context
1342  at every iteration
1343 
1344  encoder_lengths: a tensor with lengths of each encoder sequence in batch
1345  (may be None, meaning all encoder sequences are of same length)
1346 
1347  decoder_input_dim: input dimension (last dimension on decoder_inputs)
1348 
1349  decoder_state_dim: size of hidden states of LSTM
1350 
1351  attention_type: One of: AttentionType.Regular, AttentionType.Recurrent.
1352  Determines which type of attention mechanism to use.
1353 
1354  outputs_with_grads : position indices of output blobs which will receive
1355  external error gradient during backpropagation
1356 
1357  weighted_encoder_outputs: encoder outputs to be used to compute attention
1358  weights. In the basic case it's just linear transformation of
1359  encoder outputs (that the default, when weighted_encoder_outputs is None).
1360  However, it can be something more complicated - like a separate
