23 '''
24 Parallelized multi-GPU distributed trainer for Resnet 50. Can be used to train
25 on imagenet data, for example.
26 
27 To run the trainer in single-machine multi-gpu mode by setting num_shards = 1.
28 
29 To run the trainer in multi-machine multi-gpu mode with M machines,
30 run the same program on all machines, specifying num_shards = M, and
31 shard_id = a unique integer in the set [0, M-1].
32 
33 For rendezvous (the trainer processes have to know about each other),
34 you can either use a directory path that is visible to all processes
35 (e.g. NFS directory), or use a Redis instance. Use the former by
36 passing the `file_store_path` argument. Use the latter by passing the
37 `redis_host` and `redis_port` arguments.
38 '''

115 def RunEpoch(
116  args,
117  epoch,
118  train_model,
119  test_model,
120  total_batch_size,
121  num_shards,
122  expname,
123  explog,
124 ):
125  '''
126  Run one epoch of the trainer.
127  TODO: add checkpointing here.
128  '''
129  # TODO: add loading from checkpoint
130  log.info("Starting epoch {}/{}".format(epoch, args.num_epochs))
131  epoch_iters = int(args.epoch_size / total_batch_size / num_shards)
132  for i in range(epoch_iters):
133  # This timeout is required (temporarily) since CUDA-NCCL
134  # operators might deadlock when synchronizing between GPUs.
135  timeout = 600.0 if i == 0 else 60.0
136  with timeout_guard.CompleteInTimeOrDie(timeout):
137  t1 = time.time()
138  workspace.RunNet(train_model.net.Proto().name)
139  t2 = time.time()
140  dt = t2 - t1
141 
142  fmt = "Finished iteration {}/{} of epoch {} ({:.2f} images/sec)"
143  log.info(fmt.format(i + 1, epoch_iters, epoch, total_batch_size / dt))
144  prefix = "{}_{}".format(
145  train_model._device_prefix,
146  train_model._devices[0])
147  accuracy = workspace.FetchBlob(prefix + '/accuracy')
148  loss = workspace.FetchBlob(prefix + '/loss')
149  train_fmt = "Training loss: {}, accuracy: {}"
150  log.info(train_fmt.format(loss, accuracy))
151 
152  num_images = epoch * epoch_iters * total_batch_size
153  prefix = "{}_{}".format(train_model._device_prefix, train_model._devices[0])
154  accuracy = workspace.FetchBlob(prefix + '/accuracy')
155  loss = workspace.FetchBlob(prefix + '/loss')
156  learning_rate = workspace.FetchBlob(
157  data_parallel_model.GetLearningRateBlobNames(train_model)[0]
158  )
159  test_accuracy = 0
160  if (test_model is not None):
161  # Run 100 iters of testing
162  ntests = 0
163  for _ in range(0, 100):
164  workspace.RunNet(test_model.net.Proto().name)
165  for g in test_model._devices:
166  test_accuracy += np.asscalar(workspace.FetchBlob(
167  "{}_{}".format(test_model._device_prefix, g) + '/accuracy'
168  ))
169  ntests += 1
170  test_accuracy /= ntests
171  else:
172  test_accuracy = (-1)
173 
174  explog.log(
175  input_count=num_images,
176  batch_count=(i + epoch * epoch_iters),
177  additional_values={
178  'accuracy': accuracy,
179  'loss': loss,
180  'learning_rate': learning_rate,
181  'epoch': epoch,
182  'test_accuracy': test_accuracy,
183  }
184  )
185  assert loss < 40, "Exploded gradients :("
186 
187  # TODO: add checkpointing
188  return epoch + 1

191 def Train(args):
192  # Either use specified device list or generate one
193  if args.gpus is not None:
194  gpus = [int(x) for x in args.gpus.split(',')]
195  num_gpus = len(gpus)
196  else:
197  gpus = list(range(args.num_gpus))
198  num_gpus = args.num_gpus
199 
200  log.info("Running on GPUs: {}".format(gpus))
201 
202  # Verify valid batch size
203  total_batch_size = args.batch_size
204  batch_per_device = total_batch_size // num_gpus
205  assert \
206  total_batch_size % num_gpus == 0, \
207  "Number of GPUs must divide batch size"
208 
209  # Round down epoch size to closest multiple of batch size across machines
210  global_batch_size = total_batch_size * args.num_shards
211  epoch_iters = int(args.epoch_size / global_batch_size)
212  args.epoch_size = epoch_iters * global_batch_size
213  log.info("Using epoch size: {}".format(args.epoch_size))
214 
215  # Create ModelHelper object
216  train_arg_scope = {
217  'order': 'NCHW',
218  'use_cudnn': True,
219  'cudnn_exhaustice_search': True,
220  'ws_nbytes_limit': (args.cudnn_workspace_limit_mb * 1024 * 1024),
221  }
222  train_model = model_helper.ModelHelper(
223  name="resnet50", arg_scope=train_arg_scope
224  )
225 
226  num_shards = args.num_shards
227  shard_id = args.shard_id
228  if num_shards > 1:
229  # Create rendezvous for distributed computation
230  store_handler = "store_handler"
231  if args.redis_host is not None:
232  # Use Redis for rendezvous if Redis host is specified
233  workspace.RunOperatorOnce(
234  core.CreateOperator(
235  "RedisStoreHandlerCreate", [], [store_handler],
236  host=args.redis_host,
237  port=args.redis_port,
238  prefix=args.run_id,
239  )
240  )
241  else:
242  # Use filesystem for rendezvous otherwise
243  workspace.RunOperatorOnce(
244  core.CreateOperator(
245  "FileStoreHandlerCreate", [], [store_handler],
246  path=args.file_store_path,
247  prefix=args.run_id,
248  )
249  )
250  rendezvous = dict(
251  kv_handler=store_handler,
252  shard_id=shard_id,
253  num_shards=num_shards,
254  engine="GLOO",
255  exit_nets=None)
256  else:
257  rendezvous = None

259  # Model building functions
260  def create_resnet50_model_ops(model, loss_scale):
261  [softmax, loss] = resnet.create_resnet50(
262  model,
263  "data",
264  num_input_channels=args.num_channels,
265  num_labels=args.num_labels,
266  label="label",
267  no_bias=True,
268  )
269  loss = model.Scale(loss, scale=loss_scale)
270  brew.accuracy(model, [softmax, "label"], "accuracy")
271  return [loss]

277  model,

280  nesterov=1,
281  policy="step",

283  gamma=0.1
284  )
285  return opt

346  forward_pass_builder_fun=create_resnet50_model_ops,
347  param_update_builder_fun=None,

354  workspace.RunNetOnce(train_model.param_init_net)
355  workspace.CreateNet(train_model.net)
356 
357  epoch = 0
358  # load the pre-trained model and reset epoch
359  if args.load_model_path is not None:
360  LoadModel(args.load_model_path, train_model)
361 
362  # Sync the model params
363  data_parallel_model.FinalizeAfterCheckpoint(train_model)
364 
365  # reset epoch. load_model_path should end with *_X.mdl,
366  # where X is the epoch number
367  last_str = args.load_model_path.split('_')[-1]
368  if last_str.endswith('.mdl'):
369  epoch = int(last_str[:-4])
370  log.info("Reset epoch to {}".format(epoch))
371  else:
372  log.warning("The format of load_model_path doesn't match!")
373 
374  expname = "resnet50_gpu%d_b%d_L%d_lr%.2f_v2" % (
375  args.num_gpus,
376  total_batch_size,
377  args.num_labels,
378  args.base_learning_rate,
379  )
380  explog = experiment_util.ModelTrainerLog(expname, args)
381 
382  # Run the training one epoch a time
383  while epoch < args.num_epochs:
384  epoch = RunEpoch(
385  args,
386  epoch,
387  train_model,
388  test_model,
389  total_batch_size,
390  num_shards,
391  expname,
392  explog
393  )
394 
395  # Save the model for each epoch
396  SaveModel(args, train_model, epoch)
397 
398  model_path = "%s/%s_" % (
399  args.file_store_path,
400  args.save_model_name
401  )
402  # remove the saved model from the previous epoch if it exists
403  if os.path.isfile(model_path + str(epoch - 1) + ".mdl"):
404  os.remove(model_path + str(epoch - 1) + ".mdl")
